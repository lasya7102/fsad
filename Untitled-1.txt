<script>
    const matches = [
        { team1: "India", team2: "Australia", date: "2025-12-20", venue: "Wankhede Stadium" },
        { team1: "England", team2: "South Africa", date: "2025-12-22", venue: "Lord's London" },
        { team1: "Sri Lanka", team2: "Pakistan", date: "2025-12-25", venue: "Galle International Stadium" }
    ];

    const players = [
        { name: "Shubman Gill", country: "India", role: "Batsman" },
        { name: "Virat Kohli", country: "India", role: "Batsman" },
        { name: "Jasprit Bumrah", country: "India", role: "Bowler" },
        { name: "Glenn Maxwell", country: "Australia", role: "All-Rounder" }
    ];

    const venues = [
        "Wankhede Stadium, Mumbai",
        "Eden Gardens, Kolkata",
        "Melbourne Cricket Ground",
        "Lord's, London",
        "Dubai International Stadium"
    ];

    function loadMatches() {
        let html = "";
        matches.forEach(m => {
            html += `<div class='card'>
                        <b>${m.team1} vs ${m.team2}</b><br>
                        Date: ${m.date}<br>
                        Venue: ${m.venue}
                    </div>`;
        });
        document.getElementById("matches").innerHTML = html;
    }

    function loadPlayers() {
        let html = "";
        players.forEach(p => {
            html += `<div class='card'>
                        <b>${p.name}</b><br>
                        Country: ${p.country}<br>
                        Role: ${p.role}
                    </div>`;
        });
        document.getElementById("players").innerHTML = html;
    }

    function loadVenues() {
        let html = "";
        venues.forEach(v => {
            html += `<div class='card'>${v}</div>`;
        });
        document.getElementById("venues").innerHTML = html;
    }

    function showScore() {
        document.getElementById("score").innerText = "India 178/3 (18.2 overs)";
    }

    // Load everything on start
    loadMatches();
    loadPlayers();
    loadVenues();
</script>
